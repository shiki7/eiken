let selectedCats = [];

let cardCount = 20;

let deck = [];

let current = 0;

let isFlipped = false;

let results = []; // {word, ok: bool}


// ====================================================

// 初期化

window.onload = () => {

  // 星を生成

  const sc = document.getElementById('starsContainer');

  for (let i = 0; i < 60; i++) {

    const s = document.createElement('div');

    s.className = 'star';

    const size = Math.random() * 3 + 1;

    s.style.cssText = `width:${size}px;height:${size}px;left:${Math.random()*100}%;top:${Math.random()*100}%;animation-delay:${Math.random()*3}s;animation-duration:${2+Math.random()*3}s`;

    sc.appendChild(s);

  }

  buildCatGrid();

};


function buildCatGrid() {

  const grid = document.getElementById('catGrid');

  grid.innerHTML = '';

  Object.entries(CATS_META).forEach(([cat, meta]) => {

    const count = WORDS.filter(w => w.cat === cat).length;

    const btn = document.createElement('button');

    btn.className = 'cat-btn';

    btn.innerHTML = `<span class="icon">${meta.icon}</span>${cat}<span class="en-label">${meta.en} / ${count}たんご</span>`;

    btn.onclick = () => toggleCat(btn, cat);

    grid.appendChild(btn);

  });

}


function toggleCat(btn, cat) {

  if (btn.classList.contains('selected')) {

    btn.classList.remove('selected');

    selectedCats = selectedCats.filter(c => c !== cat);

  } else {

    btn.classList.add('selected');

    selectedCats.push(cat);

  }

  document.getElementById('startBtn').disabled = selectedCats.length === 0;

}


function setCount(n, btn) {

  cardCount = n;

  document.querySelectorAll('.count-btn').forEach(b => b.classList.remove('selected'));

  btn.classList.add('selected');

}

// フラッシュ開始

// ====================================================

function startFlash() {

  const pool = WORDS.filter(w => selectedCats.includes(w.cat));

  deck = shuffle([...pool]).slice(0, Math.min(cardCount, pool.length));

  current = 0;

  results = deck.map(w => ({ word: w, ok: null }));

  document.getElementById('homeScreen').classList.add('hidden');

  document.getElementById('resultScreen').style.display = 'none';

  document.getElementById('flashScreen').style.display = 'block';

  showCard();

}


function shuffle(arr) {

  for (let i = arr.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [arr[i], arr[j]] = [arr[j], arr[i]];

  }

  return arr;

}

function showCard() {

  const w = deck[current];

  isFlipped = false;


  // 表面

  document.getElementById('enWord').textContent = w.en;

  document.getElementById('enPhonetic').textContent = w.phonetic;

  document.getElementById('frontCatTag').textContent = w.cat;

  document.getElementById('catLabel').textContent = w.cat;


  // 裏面

  document.getElementById('backEn').textContent = w.en;

  document.getElementById('cardIllust').textContent = w.illust;

  document.getElementById('jaReading').textContent = w.ja;

  document.getElementById('jaDesc').textContent = w.desc;

  document.getElementById('exampleEn').textContent = w.ex;


  // カード状態リセット

  const fc = document.getElementById('flipCard');

  const fb = document.getElementById('faceBack');

  fc.classList.remove('flipped');

  fb.classList.remove('show-anim');

  document.getElementById('actionRow').classList.remove('show');


  // スコア・進捗

  const okN = results.filter(r => r.ok === true).length;

  const ngN = results.filter(r => r.ok === false).length;

  document.getElementById('okCount').textContent = okN;

  document.getElementById('ngCount').textContent = ngN;

  document.getElementById('cardNumDisp').textContent = `${current+1}/${deck.length}`;

  document.getElementById('cardNumLabel').textContent = `${current+1} まい め / ${deck.length}まい`;

  document.getElementById('progBar').style.width = `${(current / deck.length) * 100}%`;


  // 次へボタン

  document.getElementById('nextCardBtn').textContent =

    current < deck.length - 1 ? 'つぎ →' : '🎉 けっかをみる！';

}


function flipCard() {

  if (isFlipped) return;

  isFlipped = true;

  const fc = document.getElementById('flipCard');

  const fb = document.getElementById('faceBack');

  fc.classList.add('flipped');

  setTimeout(() => {

    fb.classList.add('show-anim');

    document.getElementById('actionRow').classList.add('show');

  }, 350);

}


function markCard(ok) {

  results[current].ok = ok;

  nextCard();

}


function nextCard() {

  current++;

  if (current >= deck.length) {

    showResult();

  } else {

    showCard();

  }

}


function prevCard() {

  if (current > 0) {

    current--;

    showCard();

  }

}

function showResult() {

  document.getElementById('flashScreen').style.display = 'none';

  document.getElementById('resultScreen').style.display = 'block';

  document.getElementById('progBar').style.width = '100%';


  const total = deck.length;

  const okN = results.filter(r => r.ok === true).length;

  const pct = okN / total;


  document.getElementById('resScore').textContent = okN;

  document.getElementById('resTotal').textContent = `/ ${total}まい おぼえた`;


  let emoji, stars, msg;

  if (pct >= 0.9) {

    emoji = '🏆'; stars = '⭐⭐⭐'; msg = 'かんぺき！えいごの てんさい！🎊';

  } else if (pct >= 0.7) {

    emoji = '🌟'; stars = '⭐⭐'; msg = 'すごい！よくおぼえてたね！';

  } else if (pct >= 0.5) {

    emoji = '💪'; stars = '⭐'; msg = 'がんばったね！もうすこしで かんぺき！';

  } else {

    emoji = '🌱'; stars = '🔥'; msg = 'まちがえた ところを もう いちど！';

  }

  document.getElementById('resEmoji').textContent = emoji;

  document.getElementById('resStars').textContent = stars;

  document.getElementById('resMsg').textContent = msg;


  // リスト

  const list = document.getElementById('resultList');

  list.innerHTML = '';

  results.forEach(r => {

    const w = r.word;

    const div = document.createElement('div');

    const status = r.ok === true ? 'ri-ok' : r.ok === false ? 'ri-ng' : '';

    const badge = r.ok === true ? '✅' : r.ok === false ? '❌' : '－';

    div.className = `result-item ${status}`;

    div.innerHTML = `

      <span class="ri-icon">${w.illust}</span>

      <span class="ri-en">${badge} ${w.en}</span>

      <span class="ri-ja">${w.ja}</span>

    `;

    list.appendChild(div);

  });


  // まちがいのみ再挑戦ボタン

  const ngList = results.filter(r => r.ok === false).map(r => r.word);

  const retryBtn = document.querySelector('.btn-r1');

  retryBtn.disabled = ngList.length === 0;

  retryBtn.style.opacity = ngList.length === 0 ? '0.4' : '1';

}


function retryNg() {

  const ngWords = results.filter(r => r.ok === false).map(r => r.word);

  if (ngWords.length === 0) return;

  deck = shuffle([...ngWords]);

  current = 0;

  results = deck.map(w => ({ word: w, ok: null }));

  document.getElementById('resultScreen').style.display = 'none';

  document.getElementById('flashScreen').style.display = 'block';

  showCard();

}


function goHome() {

  document.getElementById('resultScreen').style.display = 'none';

  document.getElementById('flashScreen').style.display = 'none';

  document.getElementById('homeScreen').classList.remove('hidden');

}
